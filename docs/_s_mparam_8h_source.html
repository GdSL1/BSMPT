53 /* Here is an example of the CKM Matrix given by the standard parameters. The elements V11, V23 and V33 are real in this parametrisation
54  * and are calculated by the other elements and the unitarity conditions. If the unitarity at numerical precision is not
55  * given you will end up with massive charged Goldstone bosons
56 */

58 const double theta12=13.04/180.0*M_PI;
59 const double theta13=0.201/180.0*M_PI;
60 const double theta23=2.38/180.0*M_PI;
61 const double delta=1.20;
62 
63 const double s12=std::sin(theta12);
64 const double s13=std::sin(theta13);
65 const double s23=std::sin(theta23);
66 const double c12=std::cos(theta12);
67 const double c13=std::cos(theta13);
68 const double c23=std::cos(theta23);
69 
70 const std::complex<double> imagNumber(0,1);
71 const std::complex<double> C_Vus = s12*c13;
72 const std::complex<double> C_Vub = s13*exp(-imagNumber*delta);
73 const std::complex<double> C_Vcd = s12*c23-c12*s23*s13*exp(imagNumber*delta);
74 const std::complex<double> C_Vcs = -c12*c23-s12*s23*s13*exp(imagNumber*delta);
75 const std::complex<double> C_Vtd = s12*s23-c12*c23*s13*exp(imagNumber*delta);
76 const std::complex<double> C_Vts = -c12*s23-s12*c23*s13*exp(imagNumber*delta);
77 
78 const std::complex<double> C_Vud = c12*c13;
79 const std::complex<double> C_Vtb = c23*c13;
80 const std::complex<double> C_Vcb = s23*c13;
